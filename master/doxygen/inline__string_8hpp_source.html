359 template <typename CharT, typename Traits>
360 basic_inline_string<CharT, Traits> &
361 basic_inline_string<CharT, Traits>::assign(basic_string_view<CharT, Traits> rhs)
362 {
363  auto cpop = pmemobj_pool_by_ptr(this);
364  if (nullptr == cpop)
365  throw pmem::pool_error("Invalid pool handle.");
366 
367  auto pop = pool_base(cpop);
368 
369  if (rhs.size() > capacity())
370  throw std::out_of_range("inline_string capacity exceeded.");
371 
372  auto initialized_mem =
373  (std::min)(rhs.size(), size()) + 1 /* sizeof('\0') */;
374 
375  obj::flat_transaction::run(pop, [&] {
376  detail::conditional_add_to_tx(data(), initialized_mem);
377 
378  if (rhs.size() > size())
379  detail::conditional_add_to_tx(
380  data() + initialized_mem,
381  rhs.size() - initialized_mem + 1,
382  POBJ_XADD_NO_SNAPSHOT);
383 
384  std::copy(rhs.data(),
385  rhs.data() + static_cast<ptrdiff_t>(rhs.size()),
386  data());
387  size_ = rhs.size();
388 
389  data()[static_cast<ptrdiff_t>(size_)] = '\0';
390  });
391 
392  return *this;
393 }
